symfony -q serve --allow-all-ip --no-tls --port=8000

// Déploiement

1- cons app:ep:ai;
2- Ajuster l'option -mmin (minutes) de la command find selon la fréquence de vos modifications dans /public
3- .env: APP_ENV: prod
4- Paste this in your local terminal:

clear;
echo 'Cleaning up...'
echo '' > var/log/dev.log;
rm -R var/cache/dev;
echo 'Installing dependencies...'
composer install --no-dev --optimize-autoloader;
echo 'Clearing cache...'
APP_ENV=prod APP_DEBUG=0 php bin/console cache:clear;
echo 'Compressing files...'
zip -rq Archives.zip bin composer.json composer.lock config importmap.php package.json src symfony.lock templates translations var vendor;
echo 'Compressing images...'
find './public/' -type f -mmin -37142 -exec zip -q -n .png:.webp:.jpg:.jpeg:.mp4 public.zip {} \;

echo 'Compiling assets...'
cons asset-map:compile
echo 'Compressing assets...'
zip -rq assets.zip ./public/assets;
zip -rq Archives.zip assets.zip public.zip;
rm assets.zip public.zip;
echo 'Uploading to server...'
scp /Users/iboy/Sites/mytvtime2.0/Archives.zip u109355875@access922206626.webspace-data.io:/

5- copy Archives.zip to server
6- export db
7- ionos > Espace web > Utiliser l'espace web > Connecter le domaine au dossier "/"
8- if 5, delete all tables, then import db
9- on server:

clear
echo 'Moving previous files and folders to temp directory ...'
mv bin temp/
mv config temp/
mv public/assets temp/
mv src temp/
mv templates temp/
mv translations temp/
mv var temp/
mv vendor temp/
mv composer.json temp/
mv composer.lock temp/
mv importmap.php temp/
mv package.json temp/
mv symfony.lock temp/
echo 'Unzip new files and folders...'
unzip -nq Archives.zip;
echo 'Unzip new image files in public...'
unzip -oq public.zip -d ./;
echo 'Unzip new asset files and folders...'
unzip -oq assets.zip -d ./;
echo 'Removing archive files...'
rm Archives.zip assets.zip public.zip;
echo 'Removing .DS_Store files...';
find ./ -depth -name ".DS_Store" -exec rm {} \;
echo 'Cleaning temp directory ...';
cd temp;
rm -R *;
cd ..;
echo 'Done';

11- 'Penser à connecter le domaine au dossier "/public"'
12- ionos > Espace web > Utiliser l'espace web > Connecter le domaine au dossier "/public"
13- .env: APP_ENV: dev
14- Paste in local terminal:

composer u
rm Archives.zip
